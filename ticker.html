      Loading ticker...

  <script>

    async function updateTicker() {

      try {

        const res = await fetch('https://api.coingecko.com/api/v3/simple/price?ids=bitcoin,ethereum&vs_currencies=usd');

        const fxRes = await fetch('https://api.exchangerate.host/latest?base=EUR&symbols=USD');

        const data = await res.json();

        const fx = await fxRes.json();


        const btc = data.bitcoin.usd;

        const eth = data.ethereum.usd;

        const eurusd = fx.rates.USD;

        const time = new Date().toLocaleTimeString();


        document.getElementById("ticker-text").textContent =

          `🟡 BTC: $${btc}   🔵 ETH: $${eth}   💶 EUR/USD: ${eurusd.toFixed(4)}   ⏱ ${time}     `.repeat(4);

      } catch (e) {

        document.getElementById("ticker-text").textContent = "⚠️ Error loading data...";

      }

    }


    updateTicker();

    setInterval(updateTicker, 30000); // Update every 30 seconds

  </script>
